# ============================================================
# PARSE SPSS FIXED-WIDTH LAYOUT
# Purpose:
# Read an SPSS setup file and extract variable names and
# fixed-width column positions for the ASCII .dat file.
# ============================================================

def parse_spss_layout(spss_path):
    with open(spss_path, "r", encoding="utf-8", errors="ignore") as f:
        spss_text = f.read()

    lines = spss_text.splitlines()

    start_idx = None
    for i, line in enumerate(lines):
        if line.upper().strip().startswith("DATA LIST"):
            start_idx = i
            break
    if start_idx is None:
        raise ValueError(f"Could not find DATA LIST block in {spss_path}")

    slash_idx = None
    for i in range(start_idx, min(start_idx + 1500, len(lines))):
        if "/" in lines[i]:
            slash_idx = i
            break
    if slash_idx is None:
        raise ValueError(f"Could not find '/' after DATA LIST in {spss_path}")

    end_idx = None
    for i in range(slash_idx + 1, len(lines)):
        clean = lines[i].strip()
        up = clean.upper()
        if clean == ".":
            end_idx = i
            break
        if up.startswith(("VARIABLE LABELS", "VALUE LABELS", "FORMATS", "MISSING VALUES")):
            end_idx = i
            break
    if end_idx is None:
        end_idx = len(lines)

    var_re = re.compile(r"^\s*([A-Za-z0-9_]+)\s*\(?\s*(\d+)\s*-\s*(\d+)\s*\)?\s*.*$")

    names = []
    colspecs = []

    for line in lines[slash_idx + 1:end_idx]:
        m = var_re.match(line)
        if m:
            var = m.group(1)
            start = int(m.group(2))
            end = int(m.group(3))
            if end >= start:
                names.append(var)
                colspecs.append((start - 1, end))

    return names, colspecs

# ============================================================
# LOAD ONE FIXED-WIDTH DAT FILE
# Purpose:
# Use the parsed SPSS layout to load one .dat file into pandas.
# ============================================================

def load_meps_dat(spss_path, dat_path):
    names, colspecs = parse_spss_layout(spss_path)

    df = pd.read_fwf(dat_path, colspecs=colspecs, names=names, dtype=str)
    df = df.apply(lambda col: col.astype(str).str.strip())

    # Make duplicated column names unique if needed
    if df.columns.duplicated().any():
        new_cols = []
        seen = {}
        for c in df.columns:
            if c not in seen:
                seen[c] = 0
                new_cols.append(c)
            else:
                seen[c] += 1
                new_cols.append(f"{c}__dup{seen[c]}")
        df.columns = new_cols

    return df

# ============================================================
# BUILD THE FULL POOLED ADULT NON-CANCER COHORT
# Purpose:
# Load all five longitudinal files and their paired condition
# files, restrict to adults, exclude cancer cases, and create
# standardized analysis fields for pooled modeling.
# ============================================================

all_panel_frames = []
panel_summary_rows = []

for row in pair_map:
    panel_name = row["panel_file"]
    print(f"\n==================== {panel_name} ====================")

    # Load files
    long_df = load_meps_dat(row["long_spss"], row["long_dat"])
    cond1_df = load_meps_dat(row["cond1_spss"], row["cond1_dat"])
    cond2_df = load_meps_dat(row["cond2_spss"], row["cond2_dat"])

    # Convert numeric where possible
    for c in long_df.columns:
        long_df[c] = pd.to_numeric(long_df[c], errors="ignore")

    for c in cond1_df.columns:
        cond1_df[c] = pd.to_numeric(cond1_df[c], errors="ignore")

    for c in cond2_df.columns:
        cond2_df[c] = pd.to_numeric(cond2_df[c], errors="ignore")

    # Replace MEPS negative-coded values with NaN
    for c in long_df.columns:
        if pd.api.types.is_numeric_dtype(long_df[c]):
            long_df[c] = long_df[c].replace(list(NEG_CODES), np.nan)

    for c in cond1_df.columns:
        if pd.api.types.is_numeric_dtype(cond1_df[c]):
            cond1_df[c] = cond1_df[c].replace(list(NEG_CODES), np.nan)

    for c in cond2_df.columns:
        if pd.api.types.is_numeric_dtype(cond2_df[c]):
            cond2_df[c] = cond2_df[c].replace(list(NEG_CODES), np.nan)

    cond_panel = pd.concat([cond1_df, cond2_df], ignore_index=True)

    # Adult restriction
    age_candidates = [c for c in long_df.columns if c.upper() in ["AGEY1X", "AGEY1", "AGE1X", "AGE31X", "AGE42X", "AGE53X"]]
    AGE_COL = age_candidates[0]
    long_df["AGE_BASE"] = pd.to_numeric(long_df[AGE_COL], errors="coerce")

    n_start = len(long_df)
    adult_df = long_df[long_df["AGE_BASE"] >= ADULT_AGE_MIN].copy()
    n_adult = len(adult_df)

    # Cancer exclusion using CCCODEX 11-45
    cond_panel["CCCODEX"] = pd.to_numeric(cond_panel["CCCODEX"], errors="coerce")
    cancer_ids = (
        cond_panel.loc[cond_panel["CCCODEX"].between(11, 45, inclusive="both"), "DUPERSID"]
        .dropna()
        .astype(str)
        .unique()
    )

    adult_df["DUPERSID"] = adult_df["DUPERSID"].astype(str)
    adult_df["CANCER_FLAG"] = adult_df["DUPERSID"].isin(cancer_ids).astype(int)

    n_before_cancer = len(adult_df)
    cohort_df = adult_df[adult_df["CANCER_FLAG"] == 0].copy()
    n_after_cancer = len(cohort_df)

    # Standardized analysis fields
    er_y2_candidates = [c for c in cohort_df.columns if c.upper() in ["ERTOTY2", "ERTOTY2X"]]
    totexp_y1_candidates = [c for c in cohort_df.columns if c.upper() in ["TOTEXPY1", "TOTEXPY1X"]]
    sex_candidates = [c for c in cohort_df.columns if c.upper() == "SEX"]
    race_candidates = [c for c in cohort_df.columns if c.upper() in ["RACEV1X", "RACE", "RACETHX"]]
    ins_candidates = [c for c in cohort_df.columns if c.upper() in ["INSCOVY1", "INSCOVY1X"]]
    pov_candidates = [c for c in cohort_df.columns if c.upper() in ["POVCATY1", "POVCATY1X"]]
    region_candidates = [c for c in cohort_df.columns if c.upper() in ["REGIONY1", "REGIONY1X"]]

    ER_Y2_COL = er_y2_candidates[0]
    TOTEXP_Y1_COL = totexp_y1_candidates[0]

    cohort_df["ER_Y2_COUNT"] = pd.to_numeric(cohort_df[ER_Y2_COL], errors="coerce")
    cohort_df["ER_ANY_Y2"] = (cohort_df["ER_Y2_COUNT"].fillna(0) >= 1).astype(int)

    cohort_df["TOTEXPY1_FINAL"] = pd.to_numeric(cohort_df[TOTEXP_Y1_COL], errors="coerce")
    cohort_df["LOG_TOTEXPY1"] = np.log1p(cohort_df["TOTEXPY1_FINAL"].clip(lower=0))

    if len(sex_candidates) > 0:
        cohort_df["SEX_BASE"] = cohort_df[sex_candidates[0]]
    if len(race_candidates) > 0:
        cohort_df["RACE_BASE"] = cohort_df[race_candidates[0]]
    if len(ins_candidates) > 0:
        cohort_df["INS_BASE"] = cohort_df[ins_candidates[0]]
    if len(pov_candidates) > 0:
        cohort_df["POV_BASE"] = cohort_df[pov_candidates[0]]
    if len(region_candidates) > 0:
        cohort_df["REGION_BASE"] = cohort_df[region_candidates[0]]

    cohort_df["AGE_BAND"] = pd.cut(
        cohort_df["AGE_BASE"],
        bins=[18, 35, 50, 65, 120],
        labels=["18-34", "35-49", "50-64", "65+"],
        right=False
    )

    cohort_df["FILE_CODE"] = panel_name

    print("Initial rows:", n_start)
    print("Adult rows:", n_adult)
    print("After cancer exclusion:", n_after_cancer)
    print("Outcome prevalence:", round(float(cohort_df["ER_ANY_Y2"].mean()), 4))

    panel_summary_rows.append({
        "panel_file": panel_name,
        "n_start": n_start,
        "n_adult": n_adult,
        "n_after_cancer_exclusion": n_after_cancer,
        "outcome_prevalence": round(float(cohort_df["ER_ANY_Y2"].mean()), 4)
    })

    all_panel_frames.append(cohort_df)

df_cohort_all = pd.concat(all_panel_frames, ignore_index=True)
panel_summary_df = pd.DataFrame(panel_summary_rows)

print("\nFINAL POOLED COHORT SHAPE:", df_cohort_all.shape)
display(panel_summary_df)
